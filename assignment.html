1.Difference between ‘this’ and ‘let’ and when to use what, explain with examples.

this : This Keyword useually returns the current object. The value of "this" contains based on place of useuage. For example : 

       
       class appComponent {  
           constructor(){

           }
           console.log(this);
           // It will contains all the properties of class
       }
      
       $("button").click(function(){
           console.log(this);
           // It will contains only the properties current clicked button
       });
Let : Let keyword alternative useage for "var" in ES2015/16. Compared to "var", it has many features mentioned below:
1. Block Level Scope. (The scope is available inside {} braces) Example : 

           
             {
                 let a =1;
                 var b =1;
             }
             console.log(a); // print -> Error
             console.log(b); // print -> 1
2. Declaraion Safty : Redeclaration will thorows error where as "var" will allow redeclaraion.

2. Write a function which accepts an array of strings and returns a map of character to count of the character(including spaces and special characters).

Program : 
function findCountofChars(arr) {
  let result={};
   for (let i = 0; i < arr.length; i++) { 
    for (let ii = 0; ii < arr[i].length; ii++) {
      //console.log(arr[i].charAt(ii));
       if(result[arr[i].charAt(ii)] == undefined ){
       	result[arr[i].charAt(ii)]=1;
       }else{
       	result[arr[i].charAt(ii)] +=1;
       } 
     }
    };

  return result;
}

        var inputArray = {

              d: 10
            }
          },
          e: 10,
          f: {
            g: {
              h: {

                  j: 30
                }
              }
            }
          }
        };
        var result = [];
        
         function recursiveFun(inputArray) {
          for (var key in inputArray) {
            result.push(key);
            // console.log(key, inputArray[key]);
            if (typeof(inputArray[key]) == 'object') {
              recursiveFun(inputArray[key])
            }

        }
        var finalResult = recursiveFun(inputArray);

4. Write a function to reverse every word of a string.
· Using split and reverse
· Without using split and reverse and in O(1) space (since strings are immutable, assume that the input is a character array).

Using Split & Reverse : Program : 
var givenInput="Hello World";
revertheString(givenInput);

function revertheString(input) {
  var result = '';
  if (input && input != '') {
    var arrFormat = input.split('');
    arrFormat = arrFormat.reverse();
    //console.log(arrFormat)
    for (var i = 0; i < arrFormat.length; i++) {
      result += arrFormat[i];
    }
    console.log(result);
    return result
  } else {
    alert("Please check the input")
  }

}
Output : 
    dlroW olleH
